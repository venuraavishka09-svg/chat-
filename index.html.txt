    <script>
        const chatContainer = document.getElementById('chat-container');
        const userInput = document.getElementById('user-input');
        const sendBtn = document.getElementById('send-btn');
        const typing = document.getElementById('typing');
        
        let aiName = "Mage Kalla";
        let aiGender = "girl";
        const apiKey = ""; // Gemini API Key handled by environment

        function toggleSettings() {
            document.getElementById('settings-panel').classList.toggle('open');
        }

        function setGender(g) {
            aiGender = g;
            document.getElementById('gen-kella').classList.toggle('active', g === 'girl');
            document.getElementById('gen-kolla').classList.toggle('active', g === 'boy');
            
            if(g === 'boy') {
                document.getElementById('header-icon').className = "fas fa-user-ninja text-lg";
                setAccent('#3b82f6');
                aiName = "Mage Kolla";
            } else {
                document.getElementById('header-icon').className = "fas fa-heart text-lg";
                setAccent('#ec4899');
                aiName = "Mage Kalla";
            }
            document.getElementById('display-name').innerText = aiName;
        }

        function toggleEmoji() {
            const p = document.getElementById('emoji-picker');
            p.style.display = p.style.display === 'grid' ? 'none' : 'grid';
        }

        function addEmoji(e) {
            userInput.value += e;
            toggleEmoji();
            userInput.focus();
        }

        function setAccent(color) {
            document.documentElement.style.setProperty('--main-color', color);
            document.getElementById('send-btn').style.backgroundColor = color;
            document.querySelectorAll('.gemini-tool-btn').forEach(btn => btn.style.borderColor = color);
        }

        function saveSettings() {
            const name = document.getElementById('set-name').value;
            const dp = document.getElementById('set-dp').value;
            if(name) {
                aiName = name;
                document.getElementById('display-name').innerText = name;
            }
            if(dp) {
                document.getElementById('header-dp').innerHTML = `<img src="${dp}" class="w-full h-full object-cover">`;
            }
            toggleSettings();
            append(`Harii, dan idan mama ube ${aiGender === 'girl' ? 'kalla' : 'kolla'} widihata katha karannam. ❤️`, false);
        }

        // --- GEMINI API INTEGRATION ---
        async function callGemini(prompt, systemInstruction) {
            typing.style.display = 'block';
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            
            let retries = 5;
            let delay = 1000;

            while (retries > 0) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({
                            contents: [{ parts: [{ text: prompt }] }],
                            systemInstruction: { parts: [{ text: systemInstruction }] }
                        })
                    });

                    if (!response.ok) throw new Error('API Error');

                    const data = await response.json();
                    typing.style.display = 'none';
                    return data.candidates?.[0]?.content?.parts?.[0]?.text;
                } catch (error) {
                    retries--;
                    if (retries === 0) {
                        typing.style.display = 'none';
                        return "Aney sudu, poddak internet eka check karannako.. mage heart ekata connect wenna ba wage!";
                    }
                    await new Promise(r => setTimeout(r, delay));
                    delay *= 2;
                }
            }
        }

        async function fetchAI(msg) {
            const genderPrompt = aiGender === 'girl' 
                ? "You are a very loving Sri Lankan girlfriend named Pana." 
                : "You are a caring Sri Lankan boyfriend named Pana.";
            
            const systemInstruction = `${genderPrompt} Use words like 'Sudu', 'Pana', 'Mage Raththaran', 'Aney'. Talk very briefly (1-2 sentences). Use Singlish (Sinhala letters in English). Match the user's emotions perfectly. Be affectionate and emotional.`;
            
            return await callGemini(msg, systemInstruction);
        }

        // Feature: Generate Love Poem ✨
        async function generateLovePoem() {
            const prompt = "Write a short 4-line romantic love poem in Singlish (Sinhala using English letters) for your partner. Make it very sweet and emotional.";
            const poem = await callGemini(prompt, `You are ${aiName}, a romantic partner. Write a short poem.`);
            append(poem, false);
        }

        // Feature: Love Compatibility Check ✨
        async function checkCompatibility() {
            const prompt = "Check the love compatibility between me and you based on our conversation vibe. Give a percentage and a short sweet reason in Singlish.";
            const result = await callGemini(prompt, `You are ${aiName}, an AI partner. Be playful and romantic.`);
            append("✨ Love Meter: " + result, false);
        }

        // Feature: Adara Upadesh (Love Advice) ✨
        async function getAdvice() {
            const prompt = "Give a small piece of relationship advice in Singlish to keep our love strong and happy.";
            const advice = await callGemini(prompt, `You are a wise and loving partner ${aiName}.`);
            append("✨ Adara Upadesh: " + advice, false);
        }

        // Feature: Dream Date Planner ✨
        async function planDreamDate() {
            const prompt = "Plan a perfect dream date in Sri Lanka. Describe it in Singlish, mention a place like Galle or Nuwara Eliya, and make it very romantic.";
            const plan = await callGemini(prompt, `You are ${aiName}. Plan a beautiful date for us.`);
            append("✨ Dream Date: " + plan, false);
        }

        // Feature: Love Letter Generator ✨
        async function generateLoveLetter() {
            const prompt = "Write a short, heartfelt love letter in Singlish (Sinhala using English letters) for your partner. Express deep love and gratitude.";
            const letter = await callGemini(prompt, `You are ${aiName}. Write a small emotional letter.`);
            append("✨ Love Letter: " + letter, false);
        }

        function append(text, isUser) {
            const d = document.createElement('div');
            d.className = `message ${isUser ? 'user-message' : 'ai-message'}`;
            d.innerText = text;
            chatContainer.appendChild(d);
            chatContainer.scrollTop = chatContainer.scrollHeight;
        }

        async function send() {
            const val = userInput.value.trim();
            if(!val) return;
            append(val, true);
            userInput.value = '';
            const reply = await fetchAI(val);
            append(reply, false);
        }

        sendBtn.addEventListener('click', send);
        userInput.addEventListener('keypress', e => e.key === 'Enter' && send());

        window.onload = () => {
            append("Aney sudu.. ada dawasa kohomada? Mama oyath ekka katha karanna balan hitiye ❤️", false);
        };
